var getVertex = function(intercept, slope, offset, inversionFn) { // Special cases for m at 0 and NaN if(slope === 0) return { x: intercept.x + offset * inversionFn(), y: intercept.y }; if(isNaN(slope)) return { x: intercept.x, y: intercept.y + offset * inversionFn() }; var squaredOffset = Math.pow(offset, 2.0); var deltaX = Math.sqrt(squaredOffset / (slope + 1.0)); var deltaY = deltaX / slope; return { x: intercept.x + deltaX * inversionFn(), y: intercept.y + deltaY * inversionFn() } }
